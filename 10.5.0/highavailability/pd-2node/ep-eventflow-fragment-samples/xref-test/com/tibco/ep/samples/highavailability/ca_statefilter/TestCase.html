32  import static org.junit.Assert.assertNotNull;
33  
34  import java.io.IOException;
35  import org.junit.After;
36  import org.junit.AfterClass;
37  import org.junit.Before;
38  import org.junit.BeforeClass;
39  import org.junit.Test;
40  
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  import com.streambase.sb.StreamBaseException;
45  import com.streambase.sb.unittest.SBServerManager;
46  import com.streambase.sb.unittest.ServerManagerFactory;
47  import com.tibco.ep.testing.framework.Administration;
48  import com.tibco.ep.testing.framework.Configuration;
49  import com.tibco.ep.testing.framework.ConfigurationException;
50  import com.tibco.ep.testing.framework.TransactionalDeadlockDetectedException;
51  import com.tibco.ep.testing.framework.TransactionalMemoryLeakException;
52  import com.tibco.ep.testing.framework.UnitTest;
53  
54  /**
55   * Example test case
56   */
57  public class TestCase extends UnitTest {
58  
59      private static final Logger LOGGER = LoggerFactory.getLogger(TestCase.class);
60  
61      private static SBServerManager server;
62  
63      /**
64       * Set up the server
65       *
66       * @throws StreamBaseException on start server error
67       * @throws ConfigurationException on configuration failure
68       * @throws InterruptedException on start server error
69       * @throws IOException unable to create poll directory
70       */
71      @BeforeClass
72      public static void setupServer() throws StreamBaseException, ConfigurationException, InterruptedException, IOException {
73          // create a StreamBase server and load modules once for all tests in this class
74          server = ServerManagerFactory.getEmbeddedServer();
75          server.startServer();
76          server.loadApp("com.tibco.ep.samples.highavailability.ca-statefilter-ef");
77      }
78  
79      /**
80       * Stop the server
81       *
82       * @throws StreamBaseException on shutdown failure
83       * @throws InterruptedException on shutdown failure
84       */
85      @AfterClass
86      public static void stopServer() throws InterruptedException, StreamBaseException {
87          try {
88              assertNotNull(server);
89              server.shutdownServer();
90              server = null;
91          } finally {
92              Configuration.deactiveAndRemoveAll();
93          }
94      }
95  
96      /**
97       * Start the containers
98       *
99       * @throws StreamBaseException on start container error
100      */
101     @Before
102     public void startContainers() throws StreamBaseException { 
103         // before each test, startup fresh container instances
104         server.startContainers();
105 
106         // Setup test framework before running tests
107         initialize();
108     }
109 
110     /**
111      * Display operator
112      * 
113      * Example test case that simply displays the operator
114      */
115     @Test
116     public void displayOperator() {
117         LOGGER.info("display operator");
118 
119         Administration admin = new Administration();
120 
121         LOGGER.info(admin.execute("display", "operator").toString());
122     }
123 
124     /**
125      * Stop containers
126      *
127      * @throws StreamBaseException on stop container error
128      * @throws TransactionalMemoryLeakException Leak detected
129      * @throws TransactionalDeadlockDetectedException Deadlock detected
130      */
131     @After
132     public void stopContainers() throws StreamBaseException, TransactionalMemoryLeakException, TransactionalDeadlockDetectedException {
133         // Complete test framework and check for any errors
134         complete();
135 
136         // after each test, dispose of the container instances
137         server.stopContainers();
138 
139     }
140 }
