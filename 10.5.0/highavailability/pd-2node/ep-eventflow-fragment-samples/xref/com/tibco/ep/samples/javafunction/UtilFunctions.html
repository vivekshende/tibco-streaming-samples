45  /**
46   * Class for general-purpose utility functions that deal with Things
47   */
48  public class UtilFunctions {
49  
50      /**
51       * Thing schema
52       */
53      protected static Schema ThingSchema = new Schema(null,
54              Schema.createField(DataType.STRING, "name"),
55              Schema.createField(DataType.DOUBLE, "width"),
56              Schema.createField(DataType.DOUBLE, "height"),
57              Schema.createField(DataType.DOUBLE, "depth"),
58              Schema.createField(DataType.DOUBLE, "weightKg"));
59  
60      private static Schema.Field WIDTH;
61      private static Schema.Field HEIGHT;
62      private static Schema.Field DEPTH;
63      private static Schema.Field WEIGHTKG;
64  
65      static {
66          try {
67              WIDTH = ThingSchema.getField("width");
68              HEIGHT = ThingSchema.getField("height");
69              DEPTH = ThingSchema.getField("depth");
70              WEIGHTKG = ThingSchema.getField("weightKg");
71          } catch (TupleException e) {
72              e.printStackTrace();
73          }
74      }
75  
76      /**
77       * @param things tuples
78       * @return the avg density of a bunch of Things
79       */
80      @CustomFunctionResolver("DensityCustomFunctionResolver0")
81      public static Double Density(List<Tuple> things) {
82          if(things == null || things.size() == 0)
83              return null;
84  
85          double sumDensity = 0.0;
86          int numValidThings = 0;
87          Double density = null;
88  
89          try {
90              for(Tuple t : things) {
91                  if(!(t.isNull(WIDTH) || t.isNull(HEIGHT) || t.isNull(DEPTH) || t.isNull(WEIGHTKG))) {
92                      double volume = Math.abs(t.getDouble(WIDTH) * t.getDouble(HEIGHT) * t.getDouble(DEPTH));
93                      if(!t.isNull(WEIGHTKG) && volume > 0.0) {
94                          sumDensity += t.getDouble(WEIGHTKG) / volume;
95                          ++numValidThings;
96                      }
97                  }
98              }
99  
100             if(numValidThings > 0)
101                 density = sumDensity / numValidThings;
102 
103         } catch (StreamBaseException e) {
104             throw new StreamBaseRuntimeException(e.getCause());
105         }
106 
107         return density;
108     }

132      * @return datatype

134      * @see com.streambase.sb.client.CustomFunctionResolver
135      */
136     public static CompleteDataType DensityCustomFunctionResolver0(CompleteDataType things) {
137         if(!things.equals(CompleteDataType.forList(CompleteDataType.forTuple(ThingSchema)))) {
138             throw new StreamBaseRuntimeException(new TypecheckException(MessageFormat.format("Expected list of {0}, got {1}", ThingSchema, things)));
139         }
140 
141         return CompleteDataType.forDouble();
142     }
