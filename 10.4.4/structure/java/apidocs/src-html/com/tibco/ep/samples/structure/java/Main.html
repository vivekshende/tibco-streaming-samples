030package com.tibco.ep.samples.structure.java;
031
032public class Main  {
033
034    public static void main(String []  args) {
035
036        //
037        // Application thread
038        //
039        Thread t = new Thread() {
040            public void run() {
041                System.out.println("hello from the thread");
042                try {
043                    Thread.sleep(5000);
044                } catch (InterruptedException ex) {
045                }
046            }
047        };
048        t.run();
049
050        //
051        // Wait for the thread to return before exiting main
052        //
053        try {
054             t.join();
055        } catch (InterruptedException ex) {
056             // handle interrupted exception
057        }
058
059        //
060        // Returning from main - causes the JVM to exit
061        //
062        System.out.println("returning from main");
063    }
064
065}
